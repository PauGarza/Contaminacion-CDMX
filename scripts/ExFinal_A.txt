model {
  # Likelihood
  for (i in 1:n) {
    logy[i] ~ dnorm(mu[i], tau)
    mu[i] <- alpha + beta[1]*temp[i] + beta[2]*hr[i] + beta[3]*sen_t[i] + beta[4]*cos_t[i]
  }
  
  # Priors poco informativas
  alpha ~ dnorm(0, 0.001)
  for (j in 1:4) {
    beta[j] ~ dnorm(0, 0.001)
  }
  tau ~ dgamma(0.001, 0.001)
  
  # Prediccion
  for (i in 1:n) {
    yf1[i] ~ dnorm(mu[i], tau)
  }
}
